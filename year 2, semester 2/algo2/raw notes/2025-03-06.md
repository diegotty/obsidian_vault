```python
def Padri(u, G):

def DFSr(x, G, P):
```

```python
def Cammino(u, P):
	if P[u] == -1: return []
	path = []
	while P[u] != u: #O(n)
		path.append(u)
		u = P[u]
	path.append(u) #O(1)
	path.reverse() #O(n)
	return path
```

def CamminoR(u, P):
	if P[u] == -1: return []
	if P[u] == u: return [u]
