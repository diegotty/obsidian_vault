```python
def Padri(u, G):
	n = len(G)
	P = [-1]*n
	P[u] = u
	DFSr(u, G, P)
	return P

def DFSr(x, G, P):
	for y in G[x]:
		if P[y] == -1:
			P[y] = x
			DFSr(y, G, P)
```

```python
def Cammino(u, P):
	if P[u] == -1: return []
	path = []
	while P[u] != u: #O(n)
		path.append(u)
		u = P[u]
	path.append(u) #O(1)
	path.reverse() #O(n)
	return path
```

def CamminoR(u, P):
	if P[u] == -1: return []
	if P[u] == u: return [u]
