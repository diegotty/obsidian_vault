```python
def DFS(u,M):
	###
	n = len(M)
	visitati = [0]*n
	DFSr(u, M, visitati)	
	return [x for x in range(n) if visitati[x]]
	
def DFsr(u, M, visitati):
	visitati[u] = 1
	for i in range(len(M)):
		if M[u][i] == 1 and not visitati[i]:
			DFSr(i, M, visitati)
```
